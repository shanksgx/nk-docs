## useClipboard
**useClipboard** is a Hooks for handling copied text.

### Installation package

```sh npm2yarn
npm i nk-plus-antd
```

import { Tabs } from 'nextra/components'
import ClipboardBtn from '@components/ClipboardBtn'
import SyncScrollingDemo from '@components/SyncScrollingDemo'

<Tabs items={['component', 'code']}>
  <Tabs.Tab>
    <ClipboardBtn />
  </Tabs.Tab>
  <Tabs.Tab>
    ```jsx {2, 15, 40-49} filename="ClipboardBtn.tsx"
    import React, { useState, useEffect } from 'react'
    import { useClipboard, cn } from 'nk-plus-antd'
    import { Tooltip } from 'antd'
    import { CopyIcon, SuccessIcon } from '@components/Icon'

    type propsType = {
      timeout: number
      callCopy?: (status: boolean) => void
    }

    const copyStr = 'Fast, disk space efficient package manager.'

    const ClipboardBtn: React.FC<propsType> = ({ callCopy, timeout = 2000 }) => {
      const [status, setStatus] = useState<boolean>(false)
      const { copy, reset, error, copied } = useClipboard({ timeout: timeout })

      useEffect(() => {
        if (status) {
          setTimeout(() => {
            setStatus(false)
          }, timeout)
        }
      }, [status])

      return (
        <div
        className={cn(
          'flex cursor-pointer justify-start items-center',
          status ? 'text-slate-200' : 'text-stone-600'
        )}
        >
          <span className='mx-[8px]'>{copyStr}</span>
          {status ? (
                <SuccessIcon />
              ) : (
                <Tooltip placement={'top'} title={'复制'}>
                  <div
                    id="copyBtn"
                    onClick={() => {
                      copy(copyStr)
                      setStatus(true)
                      if (copied && callCopy) {
                        callCopy(true)
                      }
                      if (error) {
                        setStatus(false)
                      }
                      reset()
                    }}
                  >
                    <CopyIcon />
                  </div>
                </Tooltip>
              )}
        </div>
      )
    }

    export default ClipboardBtn
    ```
  </Tabs.Tab>
</Tabs>

## useSyncScrolling

**useSyncScrolling** is a Hooks for Synchronize the rolling position of the box.

<Tabs items={['component', 'code']}>
  <Tabs.Tab>
    <SyncScrollingDemo />
  </Tabs.Tab>
  <Tabs.Tab>
    ```jsx {2, 5-8} filename="SyncScrollingDemo.tsx"
    import { useRef } from 'react'
    import { useSyncScrolling } from 'nk-plus-antd'

    const SyncScrollingDemo = () => {
      const docLeft = useRef<HTMLDivElement>(null)
      const docRight = useRef<HTMLDivElement>(null)

      useSyncScrolling(docLeft, docRight)

      return (
        <div className="flex">
          <div
            data-testid="left"
            ref={docLeft}
            className="bg-slate-800 text-white overflow-auto w-[500px] h-[500px]"
          >
            <div className='w-[800px] h-[800px] p-[16px]'>Left</div>
          </div>
          <div
            data-testid="right"
            className="bg-yellow-600 text-white  overflow-auto w-[500px] h-[500px]"
            ref={docRight}
          >
            <div className='w-[800px] h-[800px] p-[16px]'>right</div>
          </div>
        </div>
      )
    }

    export default SyncScrollingDemo
    ```
  </Tabs.Tab>
